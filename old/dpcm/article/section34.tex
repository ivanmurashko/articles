\subsection{Окончательный вид уравнений}

Приведем здесь окончательный вид уравнений:

\begin{eqnarray}  
\left(\frac{\partial}{\partial t}+\frac{1}{\tau} \right) 
\check N^{\ast}=QB\left(A_s^{(1)}A_c^{(1)\ast}+A_s^{(2)}A_c^{(2)\ast}\right)
,\nonumber \\ 
\left(\frac{\partial}{\partial t}+\frac 1 \tau \right) 
\check N_f^{(1)}=QC_1A_s^{(1)}A_f^{(1)\ast}
,\nonumber \\ 
 \left(\frac\partial{\partial t}+\frac 1 \tau \right) 
\check N_f^{(2)}=QC_2 A_s^{(2)}A_f^{(2)\ast}, 
,\nonumber \\ 
N=N_0 + \check N, \qquad N_f^{(1,2)}=N_{f0}^{(1,2)} + \check N_f^{(1,2)},
\label{gratings} 
\end{eqnarray} 
где 
$iM=\check N$, $-iM^\ast=\check N^\ast$, $-iM_f^{(1)}=\check N_f^{(1)}$, 
$-M_f^{(2)}=\check N_f^{(2)}$;  $Q=\frac\Gamma{2I}$; а
$N_0$ и $N_{f0}^{(1,2)}$ учитывают наличие постоянных во времени
заправочных решеток.

\begin{eqnarray}  
\frac{\partial A_{s}^{(1)}}{\partial z} = D\left\{ 
A_c^{(1)}N^{\ast} + A_f^{(1)}N_f^{(1)} \right\}   
,\nonumber \\ 
\frac{\partial A_{c}^{(1)}}{\partial z} = - D A_s^{(1)}N
,\nonumber \\ 
\frac{\partial A_{f}^{(1)}}{\partial z} = - D A_s^{(1)}N_f^{(1)\ast}, 
\label{ampl1} 
\end{eqnarray} 

\begin{eqnarray}  
\frac{\partial A_{s}^{(2)}}{\partial z} = -D\left\{ A_c^{(2)}N^\ast +
A_f^{(2)}N_f^{(2)} \right\}   
,\nonumber\\   
\frac{\partial A_{c}^{(2)}}{\partial z} =
D A_s^{(2)}N
,\nonumber\\   
\frac{\partial A_{f}^{(2)}}{\partial z} = D
A_s^{(2)}N_f^{(2)},  
\label{ampl2} 
\end{eqnarray} 
где в целях удобства полагалось, что $\theta_1 =\theta_2=\theta$, так что
$D=\frac{\omega}c  
\frac 1 {2 \cos \theta}\sqrt{\frac\mu\epsilon}$.
